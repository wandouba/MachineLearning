def performance_metric2(y_true, y_predict):
    """"""
    y_true_mean = np.mean(y_true)
    ss_tot = np.sum(np.square(y_true - y_true_mean))
    ss_res = np.sum(np.square(np.subtract(y_true, y_predict)))
    score = 1 - (ss_res/ss_tot)

    return score

    """  [X,y]"""
